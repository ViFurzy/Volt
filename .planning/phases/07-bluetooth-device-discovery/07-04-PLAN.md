    Method _on_scan_clicked(self):
      self._scan_btn.setEnabled(False)
      self._scan_btn.setText("Scanning...")
      future = asyncio.run_coroutine_threadsafe(self._service.scan_bt_devices(), self._loop)
      # Done callback runs on the bg thread — must not touch Qt widgets directly
      # The actual result arrives via _ui_queue → MonitorApp.drain() → on_scan_result()
      future.add_done_callback(lambda f: None)  # fire-and-forget; result via queue

    Method on_scan_result(self, devices: list):
      """Called by MainWindow when BtScanResultEvent arrives from drain(). Runs on Qt main thread."""
      self._scan_btn.setEnabled(True)
      self._scan_btn.setText("Scan")
      self._scan_list.clear()
      for d in devices:
          battery_str = f"{d['battery']}%" if d.get("battery") is not None else "battery unknown"
          text = f"{d['name']}  |  {battery_str}"
          item = QListWidgetItem(text)
          item.setData(Qt.ItemDataRole.UserRole, d)
          self._scan_list.addItem(item)

    Method _on_add_clicked(self):
      selected = self._scan_list.currentItem()
      if selected is None: return
      d = selected.data(Qt.ItemDataRole.UserRole)
      cfg = load_config()
      monitored = cfg.get("monitored_devices", [])
      if any(e["id"] == d["id"] for e in monitored): return  # already added
      entry = {"id": d["id"], "name": d["name"], "type": d.get("type", "bt"), "ble_address": d.get("ble_address")}
      monitored.append(entry)
      cfg["monitored_devices"] = monitored
      save_config(cfg)
      label = f"{entry['name']}  |  {entry['type'].upper()}"
      item = QListWidgetItem(label)
      item.setData(Qt.ItemDataRole.UserRole, entry)
      self._monitored_list.addItem(item)

      save_config(cfg)
      row = self._monitored_list.row(selected)

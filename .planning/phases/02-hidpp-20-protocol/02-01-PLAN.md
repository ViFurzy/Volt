      5. If len(response) < min_len: return None  (short/malformed response guard)

    ```python
    import hid, sys

    LOGITECH_VID = 0x046D
    VENDOR_USAGE_PAGE = 0xFF00

    all_devs = hid.enumerate(LOGITECH_VID, 0)
    candidates = [d for d in all_devs if d["usage_page"] == VENDOR_USAGE_PAGE]
    if not candidates:
        print("No usage_page=0xFF00 interface found")
        sys.exit(1)

    info = candidates[0]

    print(f"Opening PID=0x{info['product_id']:04X} path={path}")

    try:
        dev.open_path(path)

        msg = [0x10, 0x01, 0x00, 0x01, 0x00, 0x00, 0x00]

        resp = dev.read(20, timeout_ms=200)
        if not resp:
            print("TIMEOUT — mouse may be off or wrong device index")
        else:
            print(f"Response ({len(resp)} bytes):")
            for i, b in enumerate(resp):
                print(f"  [{i}] = 0x{b:02X} ({b})")
    finally:
        dev.close()
    ```

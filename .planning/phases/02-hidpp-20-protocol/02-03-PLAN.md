    BatteryResult:
    - percent: int (0–100; None if unknown e.g. charging via 0x1000)
    - charging: bool
    - feature_used: str (e.g. "0x1004", "0x1000", "0x1001")

    get_feature_index(device, device_idx, feature_id) -> int | None:
    - Builds GetFeature request: build_short_msg(device_idx, feature_idx=0x00, function=0, params=(feature_id>>8, feature_id&0xFF, 0))
    - Calls send_and_recv
    - Returns None on timeout (result is None)
    - Returns None if response[4+OFFSET] == 0x00 (feature absent)
    - Raises HIDppError on error responses (let send_and_recv raise through)
    - Returns int (the runtime feature index) otherwise

    battery_probe_chain(device, device_idx) -> BatteryResult | None:
    - OFFLINE guard: catches HIDppError with code==ERROR_OFFLINE anywhere in the chain and returns None
    - Tries FEAT_UNIFIED_BATTERY (0x1004): get_feature_index, then _read_1004
    - Tries FEAT_BATTERY_STATUS (0x1000): get_feature_index, then _read_1000
    - Tries FEAT_BATTERY_VOLTAGE (0x1001): get_feature_index, then _read_1001
    - Returns None if no feature responds

    _read_1004(device, device_idx, feature_idx) -> BatteryResult:
    - Calls GetCapabilities (function=0): checks response[5+OFFSET] & 0x02 for FLAG_STATE_OF_CHARGE
    - Calls GetStatus (function=1): reads percent from response[4+OFFSET] (if SOC flag set), else uses level bucket map
    - Level bucket map: {0x01: 5, 0x02: 20, 0x04: 70, 0x08: 100} — pick highest set bit from response[5+OFFSET]
    - charging: response[6+OFFSET] in (1, 2, 3)  (CHARGING, CHARGING_SLOW, CHARGE_COMPLETE)

    _read_1000(device, device_idx, feature_idx) -> BatteryResult:
    - Calls GetBatteryLevelStatus (function=0)
    - percent: response[4+OFFSET] if response[6+OFFSET] != 1 else None  (None when charging, not 0)
    - charging: response[6+OFFSET] == 1  (RECHARGING)

    _read_1001(device, device_idx, feature_idx) -> BatteryResult:
    - Calls GetBatteryVoltage (function=0)
    - voltage_mv = (response[4+OFFSET] << 8) | response[5+OFFSET]
    - percent = voltage_to_percent(voltage_mv)
    - charging: (response[6+OFFSET] & 0xE0) in (0x00, 0x20)  (Charging or Full)

    voltage_to_percent(mv: int) -> int:
    - mv >= 4200 → 100
    - 3700 <= mv < 4200 → 50 + int((mv - 3700) / 500 * 50)
    - 3000 <= mv < 3700 → int((mv - 3000) / 700 * 50)
    - mv < 3000 → 0
